# To check the Docker version
docker version
# To check more information about docker
docker info

#Access Docker filesystem
cd /var/lib/docker

#Docker images reside in
/var/lib/docker/devicemapper/imagedb/content/sha256/

# Pull from docker hub
docker pull centos:latest

#Create a docker container
docker run centos:latest /bin/echo "Hello for this container"

#Create a docker container and attach to tty or terminal
docker run -it centos:latest /bin/bash

#Create a docker container and attach to tty or terminal but run in daemon or background
docker run -itd centos:latest /bin/bash

#Attach to the container that is running in a daemon or background
docker attach <containerid | container name>


# docker restart
docker restart <containerid | container name>

#inspect content of docker container
docker inspect <containerid | container name>

# Commit a docker container
docker commit -m "message" -a "author|Maintainer" <containerid | container name> centos:v1

#Create a docker image from Dockerfile
docker build -t centos:v2 .

#execute any command on a running container as a seperate process
docker exec <containerid | container name> command

# Assign the container port to the Host OS Port(Port Binding)
docker run -d -p 80:80 nginx:latest

# Execute the command on a running container as a seperate process as a root(0 Means root user)
docker exec -u 0 -it <containerid | container name>

# Provide a name to your container using --name
docker run -it --name "testname" -v /opt/:/opt/ centos:latest /bin/bash

# Login to Docker hub
docker login
#Push your image to docker hub
docker push <ImageID>
#Logout from Docker hub
docker logout

#build image from docker file
docker build tag <tagname> <docker file path>
